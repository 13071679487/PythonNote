def deepCopy(srcDir,destDir):
    for root, dirs, files in os.walk(srcDir):
        # print(root)  # 当前目录路径
        while len(dirs) > 0:
            for dir in dirs:
                os.mkdir('{}/{}'.format(destDir, dir))
                deepCopy('{}/{}'.format(srcDir, dir), '{}/{}'.format(destDir, dir))
        for file in files:
            inFile = open('{}/{}'.format(srcDir, file), 'r', encoding='utf-8')
            outFile =open('{}/{}'.format(destDir, file), 'a', encoding='utf-8')
            notNull = True
            while notNull:
                str = inFile.readline()
                if str == '':
                    notNull = False
                    break

            inFile.close()

        print(dirs)  # 当前路径下所有子目录

5、异常
    1、概念：
        错误：由于逻辑户语法错误造成的一个程序无法正常执行的问题
        异常：是程序执行出错时的一种状态

    2、try、except语句
        格式：
            try:
                可能触发异常的语句
            except 异常类型 [as 新名]:
                异常处理语句1
            except:
                异常处理语句2
            else:
                未发生异常执行的语句
            finally:
                最终执行语句

    3、执行的流程：
        1、try语句：检测可能触发异常的语句
        2、except语句：可以有多条，但至少有一条
        3、else语句：可以省略
        4、finally语句：可以省略，最终执行语句一定会指定（放到最后）

        例：
            def f1():
                s = input('请输入：')
                return s

            try:
                value = f1()
                n = 10 / int(value)
            except ZeroDivisionError:
                print('分母不能为0')
            except ValueError:
                print('输入的内容必须是数字')
            else:
                print(n)
            finally:
                print('程序结束')

    4、常见的异常类型：
